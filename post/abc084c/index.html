問題記載のとおり, ダイアグラムを考え実装するだけ。 自力で実装したが, 解説を見たらそちらの方がわかりやすかった。

以下提出コードです。

fn read<T: std::str::FromStr>() -> T {
    let mut s = String::new();
    std::io::stdin().read_line(&mut s).ok();
    s.trim().parse().ok().unwrap()
}

fn read_vec<T: std::str::FromStr>() -> Vec<T> {
    read::<String>().split_whitespace()
        .map(|e| e.parse().ok().unwrap()).collect()
}

fn main() {
    let n: usize = read::<usize>();
    let mut v: Vec<Vec<u64>> = Vec::new();
    let mut t: Vec<u64> = Vec::new();

    for i in 0..n-1 {
        let tmp: Vec<u64> = read_vec::<u64>();
        // 次駅までにかかる時刻, 始発時刻, 発射間隔
        v.push(tmp);
    }

    for i in 0..n-1 {
        let mut tmp: u64 = v[i][0]+v[i][1];
        for j in i+1..n-1 {
            if tmp > v[j][1] {
                let mut tmpt: u64 = v[j][1];
                while tmp > tmpt {
                    tmpt += v[j][2];
                }
                tmp = tmpt;
            } else {
                tmp = v[j][1];
            }
            tmp += v[j][0];
        }
        t.push(tmp);
    }

    t.push(0);
    for i in 0..t.len() {
        println!("{}", t[i]);
    }
}
